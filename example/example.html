1.8.6 Horizontal Spacing

\[ \begin{align*} f(x) &= x^2\! +3x\! +2 \\ f(x) &= x^2+3x+2 \\ f(x) &= x^2\, +3x\, +2 \\ f(x) &= x^2\: +3x\: +2 \\ f(x) &= x^2\; +3x\; +2 \\ f(x) &= x^2\ +3x\ +2 \\ f(x) &= x^2\quad +3x\quad +2 \\ f(x) &= x^2\qquad +3x\qquad +2 \end{align*} \]

1.8.7 Vertical Spacing

\[ \begin{aligned} T_{P} &= K_{T}. \rho . n^{2}_{p} . D^{4}_{p} \\[1pt] Q_{P} &= K_{Q}. \rho . n^{2}_{p} . D^{5}_{p} \\[10pt] N_{P} &= K_{N}. \rho . n^{2}_{p} . D^{6}_{p} \\[20pt] K_{P} &= K_{K}. \rho . n^{2}_{p} . D^{7}_{p} \end{aligned} \]
